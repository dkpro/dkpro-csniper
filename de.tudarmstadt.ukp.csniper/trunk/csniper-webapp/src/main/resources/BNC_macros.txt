# !NP: Nominalphrase (sehr vereinfacht)
MACRO NP(0)
(
	[pos="AT0"]? 
	[]? 
	[pos="AJ.*"]* 
	[pos="N.*"]
)
;

MACRO PP(0)

# !VCF: finite verbale Kette
MACRO VCF(0)
(
	[pos="V.?B" | pos="V.?D" | pos="V.?Z" | pos="VM0"]
	[pos="V.*"]*
)
;

)
;

# !RC: Relativsatz
MACRO RC(0)
(
	([pos="DTQ" | pos="PNQ" | pos="CJT"] /VCF[] []?) |
	([pos="CJT"]? /NP[] /VCF[] []?) |
	([pos="PR.*"]* [pos=".Q"] /NP[] /VCF[] []?)
)
;
